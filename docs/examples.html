Examples

Three complete projects you can clone and run today. All available in framework/examples/.

Smart Home Controller

An ESP32 or Raspberry Pi with a BME280 sensor, light relay, and fan relay. Control everything from Telegram with natural language.

You:is it too hot? turn on the fan if needed
WISP:🌡 temperature: 29.4 °C — that's warm. Turning fan on.
✅ fan: on
You:dim the lights to 30%
WISP:✅ led_brightness: 30
# examples/smart_home/main.py
from wisp import WispDevice, capability

class SmartHome(WispDevice):
    device_name = "smart_home"
    device_description = "Home controller with temperature, fan, and LED."

    @capability
    def read_temperature(self) -> dict:
        """Read temperature and humidity from BME280."""
        sensor = self.hardware.sensor("bme280")
        return {
            "temperature": round(sensor.temperature, 2),
            "humidity": round(sensor.humidity, 2),
            "unit": "C",
        }

    @capability
    def set_fan(self, state: str) -> dict:
        """Turn the ceiling fan on or off. state: 'on' or 'off'."""
        self.hardware.set_output("fan", state == "on")
        return {"fan": state}

    @capability
    def set_led_brightness(self, brightness: int) -> dict:
        """Set LED brightness 0-100."""
        self.hardware.pwm("led", brightness)
        return {"led_brightness": brightness}

    @capability
    def all_off(self) -> dict:
        """Turn off all outputs."""
        self.hardware.set_output("fan", False)
        self.hardware.set_output("led", False)
        return {"fan": "off", "led": "off"}


SmartHome.from_config("config.json").run()

Weather Station

A Raspberry Pi with BME280, BH1750, and SHT31 sensors. Ask for any reading in plain English. Historical logging to a CSV file.

You:is it bright enough to work without lights?
WISP:☀️ light: 420 lux — yes, plenty of natural light.
You:log all sensors now
WISP:✅ Logged: temp=22.1°C, humidity=58%, lux=420
# examples/weather_station/main.py
import csv, time
from wisp import WispDevice, capability

class WeatherStation(WispDevice):
    device_name = "weather_station"
    device_description = "Outdoor weather station with multiple sensors."
    LOG_FILE = "log.csv"

    @capability
    def read_all(self) -> dict:
        """Read temperature, humidity, pressure, and light level."""
        bme = self.hardware.sensor("bme280")
        lux = self.hardware.sensor("bh1750")
        return {
            "temperature": round(bme.temperature, 2),
            "humidity": round(bme.humidity, 2),
            "pressure": round(bme.pressure, 1),
            "lux": round(lux.lux, 1),
        }

    @capability
    def log_reading(self) -> dict:
        """Log the current sensor reading to a CSV file."""
        data = self.read_all()
        data["timestamp"] = time.strftime("%Y-%m-%dT%H:%M:%S")
        with open(self.LOG_FILE, "a", newline="") as f:
            csv.DictWriter(f, data.keys()).writerow(data)
        return {"logged": True, **data}


WeatherStation.from_config("config.json").run()

ROS2 Robot

Control a TurtleBot3 from Telegram — movement, sensors, navigation — with zero custom capability code. ROS2Plugin does it all.

You:move forward for 2 seconds then stop
WISP:🤖 Moving forward... ✅ stopped.
You:what's the battery level?
WISP:🔋 battery: 87%
You:navigate to coordinates 2.0, 1.5
WISP:🗺 Nav2 goal sent: x=2.0, y=1.5
# examples/ros2_robot/main.py
import rclpy
from wisp import WispDevice
from wisp.plugins.ros2 import ROS2Plugin

class TurtleBot(WispDevice):
    device_name = "turtlebot3"
    device_description = "TurtleBot3 Burger mobile robot."
    # No @capability needed — ROS2Plugin discovers everything

rclpy.init()
device = TurtleBot.from_config("config.json")
device.use(ROS2Plugin())
device.run()

{

  },

    "token": "your_bot_token",
    "allowed_users": ["your_username"]
  }
}
